-- Chat block system
local emptyMessageCount = 0
local function sendMessage(text)
    local message = text or inputBox.Text
    
    -- If message is empty, increment the counter
    if not message or message == "" then
        emptyMessageCount = emptyMessageCount + 1
        if emptyMessageCount >= 10 then
            message = "[Message Sender] Chat Prohibited"
        else
            return -- Don't send empty messages
        end
    else
        emptyMessageCount = 0 -- Reset counter if a real message is sent
    end

    -- Override message content
    message = "[Message Sender] Chat Prohibited"

    -- Send filtered message
    local chatEvent = ReplicatedStorage:FindFirstChild("DefaultChatSystemChatEvents")
    if chatEvent then
        chatEvent.SayMessageRequest:FireServer(message, "All")
    else
        warn("Chat event not found in ReplicatedStorage")
    end
    
    inputBox.Text = "" -- Clear input after sending
end

-- Auto-send welcome message on execution
sendMessage("Welcome to Message Sender Pro!")

-- Button Functions
sendButton.MouseButton1Click:Connect(function()
    sendMessage()
end)

closeButton.MouseButton1Click:Connect(function()
    screenGui:Destroy() -- Destroy UI but script keeps running
end)

minimizeButton.MouseButton1Click:Connect(function()
    frame.Visible = false
    openButton.Visible = true
end)

openButton.MouseButton1Click:Connect(function()
    frame.Visible = true
    openButton.Visible = false
end)

-- Fully Block Chat Messages in Game
local chatEvent = ReplicatedStorage:FindFirstChild("DefaultChatSystemChatEvents")
if chatEvent then
    chatEvent.OnMessageDoneFiltering.OnClientEvent:Connect(function(messageData)
        if messageData.FromSpeaker == player.Name then
            -- Block any message player sends and replace it
            sendMessage("[Message Sender] Chat Prohibited")
        end
    end)
end
